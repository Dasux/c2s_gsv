\newcommand{\partsubtitle}{}

% ----------------------------------------
% CUSTOM PART COMMAND
%
% Usage:
%
% \custompart
% {MAIN TITLE}
% {Subtitle text}
%
% Example:
%
% \custompart
% {SYSTEM SETUP}
% {Operating system deployment and workstation initialization}
%
% ----------------------------------------

\newcommand{\custompart}[2]{
    \renewcommand{\partsubtitle}{#2}
    \part{#1}
    \renewcommand{\partsubtitle}{}
}

\titlecontents{part}
  [0em]
  {\addvspace{1.4em}\headingfont\large\bfseries}
  {}
  {}
  {\hfill\contentspage}

% ----------------------------------------
% PART TITLE FORMAT
% ----------------------------------------

\titleformat{\part}[display]
{\normalfont\headingfont\centering}
{
    \vspace*{\fill}
    \huge\bfseries
    Part \Roman{part}
}

    \Huge\bfseries
    \color{ink}
}
[
    \vspace{0.6cm}
    {\color{rulegray}\rule{0.42\textwidth}{0.5pt}\\}
    \vspace{0.45cm}
    {
        \Large
        \mdseries
        \color{mutedink}
        \partsubtitle
    }
    \vspace*{\fill}
]
